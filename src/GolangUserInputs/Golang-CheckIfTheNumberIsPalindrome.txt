package main

import "fmt"

func main() {

	var originalNumber int = 121
	var reverseNumber int = 0
	var temp = originalNumber

	for temp > 0 {
		var remainder = temp % 10
		reverseNumber1 := reverseNumber * 10
		reversenumber2 := reverseNumber1 + remainder
		reverseNumber = reversenumber2
		temp = temp / 10
	}

	if originalNumber == reverseNumber {
		fmt.Print("1")
	} else {
		fmt.Print("-1")
	}
}
